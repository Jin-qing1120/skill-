JavaScript基础知识练习

1. 变量定义

// 使用let定义变量
let name = "JavaScript";
// 使用const定义常量
const version = 1.0;
// 使用var定义变量（旧方式）
var age = 25;

console.log("Name:", name);
console.log("Version:", version);
console.log("Age:", age);

2. 变量类型

// 字符串类型
let stringType = "Hello World";
// 数字类型
let numberType = 42;
// 布尔类型
let booleanType = true;
// 数组类型
let arrayType = [1, 2, 3, 4, 5];
// 对象类型
let objectType = { name: "John", age: 30 };
// 函数类型
let functionType = function() { return "Hello"; };
// undefined类型
let undefinedType;
// null类型
let nullType = null;

console.log("String type:", typeof stringType);
console.log("Number type:", typeof numberType);
console.log("Boolean type:", typeof booleanType);
console.log("Array type:", typeof arrayType);
console.log("Object type:", typeof objectType);
console.log("Function type:", typeof functionType);
console.log("Undefined type:", typeof undefinedType);
console.log("Null type:", typeof nullType);

3. 表达式

// 算术表达式
let sum = 10 + 5;
let difference = 10 - 5;
let product = 10 * 5;
let quotient = 10 / 5;
let remainder = 10 % 3;

// 比较表达式
let isEqual = 10 == "10";
let isStrictEqual = 10 === "10";
let isGreater = 10 > 5;
let isLess = 10 < 5;

// 逻辑表达式
let andExpression = true && false;
let orExpression = true || false;
let notExpression = !true;

console.log("Sum:", sum);
console.log("Difference:", difference);
console.log("Product:", product);
console.log("Quotient:", quotient);
console.log("Remainder:", remainder);
console.log("Is equal:", isEqual);
console.log("Is strict equal:", isStrictEqual);
console.log("Is greater:", isGreater);
console.log("Is less:", isLess);
console.log("And expression:", andExpression);
console.log("Or expression:", orExpression);
console.log("Not expression:", notExpression);

4. 流程控制

// if-else语句
let score = 85;
if (score >= 90) {
    console.log("优秀");
} else if (score >= 80) {
    console.log("良好");
} else if (score >= 60) {
    console.log("及格");
} else {
    console.log("不及格");
}

// switch语句
let day = 3;
switch (day) {
    case 1:
        console.log("星期一");
        break;
    case 2:
        console.log("星期二");
        break;
    case 3:
        console.log("星期三");
        break;
    case 4:
        console.log("星期四");
        break;
    case 5:
        console.log("星期五");
        break;
    default:
        console.log("周末");
}

// for循环
console.log("For loop:");
for (let i = 0; i < 5; i++) {
    console.log(i);
}

// while循环
console.log("While loop:");
let j = 0;
while (j < 5) {
    console.log(j);
    j++;
}

// do-while循环
console.log("Do-while loop:");
let k = 0;
do {
    console.log(k);
    k++;
} while (k < 5);

5. 函数

// 函数声明
function greet(name) {
    return "Hello, " + name + "!";
}

// 函数表达式
const calculateSum = function(a, b) {
    return a + b;
};

// 箭头函数
const multiply = (a, b) => a * b;

// 带默认参数的函数
function greetWithDefault(name = "World") {
    return "Hello, " + name + "!";
}

// 调用函数
console.log(greet("John"));
console.log(calculateSum(5, 3));
console.log(multiply(4, 6));
console.log(greetWithDefault());
console.log(greetWithDefault("Alice"));
